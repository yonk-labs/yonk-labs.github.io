{{- $page := .Page -}} {{- $captions := $page.Params.gallery | default slice -}} {{- /* Build a map from filename to caption metadata */ -}} {{- $captionMap := dict -}} {{- range $captions -}} {{- $captionMap = merge $captionMap (dict .file .) -}} {{- end -}}
{{- range $page.Resources.Match "*.jpg" -}} {{- $name := path.Base .Name -}} {{- $meta := index $captionMap $name -}} {{- $title := "" -}} {{- $caption := "" -}} {{- with $meta -}} {{- $title = .title | default "" -}} {{- $caption = .caption | default "" -}} {{- end -}} {{- if not $title -}} {{- $title = replace (replaceRE "\\.[^.]+$" "" $name) "_" " " | replaceRE "([a-z])([A-Z])" "$1 $2" | title -}} {{- end -}} {{- end }} {{- range $page.Resources.Match "*.png" -}} {{- $name := path.Base .Name -}} {{- $meta := index $captionMap $name -}} {{- $title := "" -}} {{- $caption := "" -}} {{- with $meta -}} {{- $title = .title | default "" -}} {{- $caption = .caption | default "" -}} {{- end -}} {{- if not $title -}} {{- $title = replace (replaceRE "\\.[^.]+$" "" $name) "_" " " | replaceRE "([a-z])([A-Z])" "$1 $2" | title -}} {{- end -}} {{- end }} {{- range $page.Resources.Match "*.webp" -}} {{- $name := path.Base .Name -}} {{- $meta := index $captionMap $name -}} {{- $title := "" -}} {{- $caption := "" -}} {{- with $meta -}} {{- $title = .title | default "" -}} {{- $caption = .caption | default "" -}} {{- end -}} {{- if not $title -}} {{- $title = replace (replaceRE "\\.[^.]+$" "" $name) "_" " " | replaceRE "([a-z])([A-Z])" "$1 $2" | title -}} {{- end -}} {{- end }}
